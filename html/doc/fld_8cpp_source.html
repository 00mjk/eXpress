00021 FLD::FLD(double alpha, size_t max_val, size_t mean, size_t std_dev) : 
00022     _hist(max_val+1),
00023     _tot_mass(HUGE_VAL), 
00024     _sum(HUGE_VAL),
00025     _min(max_val)
00026 {
00027     assert(KERNEL.size() % 2 == 1);
00028     boost::math::normal norm(mean, std_dev);
00029     
00030     double tot = log(alpha*max_val);
00031     
00032     for (size_t i = 1; i <= max_val; ++i)
00033     {  
00034         double mass = tot + log(boost::math::cdf(norm,i+0.5) - boost::math::cdf(norm,i-0.5));
00035         _hist[i] = mass;
00036         _sum = log_sum(_sum, log((double)i)+mass);
00037         _tot_mass = log_sum(_tot_mass, mass);
00038     }

00047 {
00048     if (len > max_val()) return;
00049     if (len < _min) _min = len;
00050     
00051     size_t offset = len - KERNEL.size()/2; 
00052     
00053     for (size_t i = 0; i < KERNEL.size(); i++)
00054     {
00055         double k_mass = mass + KERNEL[i];
00056         _hist[offset] = log_sum(_hist[offset], k_mass);
