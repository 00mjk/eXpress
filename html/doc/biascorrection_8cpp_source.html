00302 double BiasBoss::get_transcript_bias(std::vector<double>& start_bias, std::vector<double>& end_bias, const Transcript& trans) const
00303 {
00304     double tot_start = HUGE_VAL;
00305     double tot_end = HUGE_VAL;
00306     
00307     size_t l = upper_bound(_5_pos_bias.len_bins().begin(),_5_pos_bias.len_bins().end(), trans.length()) - _5_pos_bias.len_bins().begin();
00308     
00309     size_t p = 0;
00310     double next_bin_start = trans.length() * _5_pos_bias.pos_bins()[p];
00311     double curr_5_pos_bias = _5_pos_bias.get_weight(l,p);
00312     double curr_3_pos_bias = _3_pos_bias.get_weight(l,p);
00313     for (size_t i = 0; i < trans.length(); ++i)
00314     {
00315         if (i >= next_bin_start)
00316         {
00317             next_bin_start = trans.length() * _5_pos_bias.pos_bins()[++p];
00318             curr_5_pos_bias = _5_pos_bias.get_weight(l,p);
00319             curr_3_pos_bias = _3_pos_bias.get_weight(l,p);
00320         }
00321         start_bias[i] = _5_seq_bias.get_weight(trans.seq(), (int)i) + curr_5_pos_bias;
00322         end_bias[i] = _3_seq_bias.get_weight(trans.seq(), (int)i) + curr_3_pos_bias;
00323         tot_start = log_sum(tot_start, start_bias[i]);
00324         tot_end = log_sum(tot_start, end_bias[i]);
00325     }
00326     
00327     double avg_bias = (tot_start + tot_end) - (2*log((double)trans.length()));
00328     return avg_bias;
