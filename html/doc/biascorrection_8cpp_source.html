00294 double BiasBoss::get_target_bias(std::vector<float>& start_bias, std::vector<float>& end_bias, const Target& targ) const
00295 {
00296     double tot_start = HUGE_VAL;
00297     double tot_end = HUGE_VAL;
00298     
00299     const Sequence& t_seq_fwd = targ.seq(0);
00300     const Sequence& t_seq_rev = targ.seq(1);
00301 
00302     for (size_t i = 0; i < targ.length(); ++i)
00303     {
00304         start_bias[i] = _5_seq_bias.get_weight(t_seq_fwd, i);// + curr_5_pos_bias;
00305         end_bias[targ.length()-i-1] = _3_seq_bias.get_weight(t_seq_rev, i);// + curr_3_pos_bias;
00306         tot_start = log_sum(tot_start, start_bias[i]);
00307         tot_end = log_sum(tot_start, end_bias[i]);
00308     }
00309     
00310     double avg_bias = (tot_start + tot_end) - (2*log((double)targ.length()));
00311     assert(!isnan(avg_bias));
00312     return avg_bias;
00313 }
