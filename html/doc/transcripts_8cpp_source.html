00301 {
00302     vector<bool> polytope_bound(bundle_trans.size(), false);
00303     
00304     while (true)
00305     {
00306         double unbound_counts = 0;
00307         double bound_counts = 0;
00308         for (size_t i = 0; i < bundle_trans.size(); ++i)
00309         {
00310             Transcript& trans = *bundle_trans[i];
00311             
00312             if (trans_counts[i] > trans.tot_counts())
00313             {
00314                 trans_counts[i] = trans.tot_counts();
00315                 polytope_bound[i] = true;
00316             }
00317             else if (trans_counts[i] < trans.uniq_counts())
00318             {
00319                 trans_counts[i] = trans.uniq_counts();
00320                 polytope_bound[i] = true;
00321             }
00322             
00323             if (polytope_bound[i])
00324             {
00325                 bound_counts += trans_counts[i];
00326             }
00327             else
00328             {
00329                 unbound_counts += trans_counts[i];
00330             }
00331         }
00332         
00333         if (unbound_counts + bound_counts == bundle_counts)
00334             return;
00335         
00336         double normalizer = (bundle_counts - bound_counts)/unbound_counts;
00337         bool unbound_exist = false;
00338         unbound_counts = 0;
00339         for (size_t i = 0; i < bundle_trans.size(); ++i)
00340         {    
00341             if (!polytope_bound[i])
00342             {
00343                 trans_counts[i] *= normalizer;
00344                 unbound_counts += trans_counts[i];
00345                 unbound_exist = true;
00346             }
00347         }
00348         
00349         if (unbound_counts + bound_counts - bundle_counts < EPSILON)
00350             return;
00351         
00352         if (!unbound_exist)
00353             polytope_bound = vector<bool>(bundle_trans.size(), false);
