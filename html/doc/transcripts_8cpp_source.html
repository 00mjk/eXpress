00256 void project_to_polytope(vector<Transcript*> bundle_trans, vector<double>& trans_counts, double bundle_counts)
00257 {
00258     vector<bool> polytope_bound(bundle_trans.size(), false);
00259     
00260     while (true)
00261     {
00262         double unbound_counts = 0;
00263         double bound_counts = 0;
00264         for (size_t i = 0; i < bundle_trans.size(); ++i)
00265         {
00266             Transcript& trans = *bundle_trans[i];
00267             
00268             if (trans_counts[i] > trans.tot_counts())
00269             {
00270                 trans_counts[i] = trans.tot_counts();
00271                 polytope_bound[i] = true;
00272             }
00273             else if (trans_counts[i] < trans.uniq_counts())
00274             {
00275                 trans_counts[i] = trans.uniq_counts();
00276                 polytope_bound[i] = true;
00277             }
00278             
00279             if (polytope_bound[i])
00280             {
00281                 bound_counts += trans_counts[i];
00282             }
00283             else
00284             {
00285                 unbound_counts += trans_counts[i];
00286             }
00287         }
00288         
00289         if (unbound_counts + bound_counts == bundle_counts)
00290             return;
00291         
00292         double normalizer = (bundle_counts - bound_counts)/unbound_counts;
00293         bool unbound_exist = false;
00294         unbound_counts = 0;
00295         for (size_t i = 0; i < bundle_trans.size(); ++i)
00296         {    
00297             if (!polytope_bound[i])
00298             {
00299                 trans_counts[i] *= normalizer;
00300                 unbound_counts += trans_counts[i];
00301                 unbound_exist = true;
00302             }
00303         }
00304         
00305         if (unbound_counts + bound_counts - bundle_counts < EPSILON)
00306             return;
00307         
00308         if (!unbound_exist)
00309             polytope_bound = vector<bool>(bundle_trans.size(), false);
00310     }
