00175 void process_fragment(double mass_n, Fragment* frag_p, FLD* fld, BiasBoss* bias_table, MismatchTable* mismatch_table, TranscriptTable* trans_table)
00176 {
00177     Fragment& frag = *frag_p;
00178     
00179     assert(frag.num_hits());
00180     
00181     frag.hits()[0]->mapped_trans->incr_bundle_counts();
00182     
00183     if (frag.num_hits()==1)
00184     // hits to a single location
00185     {
00186         const FragHit& m = *frag.hits()[0];
00187         Transcript* t  = m.mapped_trans;
00188         if (!t)
00189         {
00190             cerr << "ERROR: Transcript " << m.name << " not found in reference fasta.";
00191             exit(1);
00192         }
00193         
00194         //update parameters
00195         t->add_mass(0, mass_n);
00196         t->incr_uniq_counts();
00197         if (m.pair_status() == PAIRED)
00198                 fld->add_val(m.length(), mass_n);
00199         
00200         if (bias_table)
00201             bias_table->update_observed(m, mass_n - t->mass() + log((double)t->unbiased_effective_length()));
00202         if (mismatch_table)
00203             mismatch_table->update(m, mass_n);
00204        
00205         delete frag_p;
00206         return;
00207     }
00208     
00209     // hits to multiple locations
00210     
00211     // calculate marginal likelihoods
00212     vector<double> likelihoods(frag.num_hits());
00213     double total_likelihood = 0.0;
00214     for(size_t i = 0; i < frag.num_hits(); ++i)
00215     {
00216         const FragHit& m = *frag.hits()[i];
00217         Transcript* t = m.mapped_trans;
00218         likelihoods[i] = t->log_likelihood(m);
00219         total_likelihood = (i) ? log_sum(total_likelihood, likelihoods[i]):likelihoods[i];
00220     }
00221     if (sexp(total_likelihood) == 0)
00222     {
00223         delete frag_p;
00224         return;
00225     }
00226 
00227     // normalize marginal likelihoods
00228     TransID bundle_rep = trans_table->get_trans_rep(frag.hits()[0]->trans_id);
00229     for(size_t i = 0; i < frag.num_hits(); ++i)
00230     {
00231         const FragHit& m = *frag.hits()[i];
00232         Transcript* t  = m.mapped_trans;
00233         double p = likelihoods[i]-total_likelihood;
00234         double mass_t = mass_n + p;
00235         
00236         if (i > 0)
00237         {
00238             TransID m_rep = trans_table->get_trans_rep(m.trans_id);
00239             if (m_rep != bundle_rep)
00240                 bundle_rep = trans_table->merge_bundles(bundle_rep, m_rep);
00241         }
00242         
00243         if (sexp(p) == 0)
00244             continue;
00245         
00246         // update parameters
00247         t->add_mass(p, mass_n);
00248         
00249         if (m.pair_status() == PAIRED)
00250             fld->add_val(m.length(), mass_t);
00251  
00252         if (bias_table)
00253             bias_table->update_observed(m, mass_t - t->mass() + log((double)t->unbiased_effective_length()));
00254         if (mismatch_table)
00255             mismatch_table->update(m, mass_t);
00256         
00257         if (calc_covar)
00258         {
00259             for(size_t j = i+1; j < frag.num_hits(); ++j)
00260             {
00261                 const FragHit& m2 = *frag.hits()[j];
00262                 double p2 = likelihoods[j]-total_likelihood;
00263                 if (sexp(p2) == 0)
00264                     continue;
00265                 
00266                 double covar = 2*mass_n + p + p2;
00267                 trans_table -> update_covar(m.trans_id, m2.trans_id, covar); 
00268             }
00269         }
00270     }
00271         
00272     delete frag_p;
00273 }
