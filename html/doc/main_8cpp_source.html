00190 void process_fragment(double mass_n, Fragment* frag_p, TranscriptTable* trans_table, const Globals& globs)
00191 {
00192     Fragment& frag = *frag_p;
00193     
00194     assert(frag.num_hits());
00195         
00196     // calculate marginal likelihoods
00197     vector<double> likelihoods(frag.num_hits(),0);
00198     double total_likelihood = HUGE_VAL;
00199     
00200     if (frag.num_hits()>1)
00201     {
00202         for(size_t i = 0; i < frag.num_hits(); ++i)
00203         {
00204             const FragHit& m = *frag.hits()[i];
00205             Transcript* t = m.mapped_trans;
00206             likelihoods[i] = t->log_likelihood(m);
00207             total_likelihood = log_sum(total_likelihood, likelihoods[i]);
00208         }
00209     }
00210     else
00211     {
00212         total_likelihood = likelihoods[0];
00213     }
00214     
00215     assert(!islzero(total_likelihood));
00216     
00217     // merge bundles
00218     Bundle* bundle = frag.hits()[0]->mapped_trans->bundle();
00219     if (iteration != FIRST)
00220     {
00221         bundle->incr_counts();
00222     }
00223     for(size_t i = 1; i < frag.num_hits(); ++i)
00224     {
00225         bundle = trans_table->merge_bundles(bundle, frag.hits()[i]->mapped_trans->bundle());
00226     }
00227 
00228     // normalize marginal likelihoods
00229     for(size_t i = 0; i < frag.num_hits(); ++i)
00230     {
00231         FragHit& m = *frag.hits()[i];
00232         Transcript* t  = m.mapped_trans;
00233         double p = likelihoods[i]-total_likelihood;
00234         double mass_t = mass_n + p;
00235         
00236         assert(!(isnan(mass_t)||isinf(mass_t)));
00237         
00238         m.probability = sexp(p);
00239         
00240         assert(!isinf(m.probability));
00241         
00242         // update parameters
00243         if (iteration != LAST)
00244         {
00245             if (frag.num_hits()==1)
00246                 t->incr_uniq_counts();
00247 
00248             t->add_mass(p, mass_n);
00249             
00250             if (m.pair_status() == PAIRED)
00251                 (globs.fld)->add_val(m.length(), mass_t);
00252             if (globs.bias_table)
00253                 (globs.bias_table)->update_observed(m, mass_t - t->mass() + t->cached_effective_length());
00254             if (globs.mismatch_table)
00255                 (globs.mismatch_table)->update(m, mass_t);
00256         }
00257         else
00258         {
00259             t->add_prob_count(m.probability);
00260         }
00261         
00262         if (calc_covar && (iteration != FIRST))
00263         {
00264             for(size_t j = i+1; j < frag.num_hits(); ++j)
00265             {
00266                 const FragHit& m2 = *frag.hits()[j];
00267                 double p2 = likelihoods[j]-total_likelihood;
00268                 if (sexp(p2) == 0)
00269                     continue;
00270                 
00271                 double covar = p + p2;
00272                 if (iteration == ONLY)
00273                     covar += 2*mass_n;
00274                 
00275                 trans_table->update_covar(m.trans_id, m2.trans_id, covar); 
00276             }
00277         }
00278     }
00279 }
