00276     
00277     
00278     boost::unordered_set<Target*> targ_set;
00279 
00280     if (frag.num_hits()>1)
00281     {
00282         for(size_t i = 0; i < frag.num_hits(); ++i)
00283         {
00284             const FragHit& m = *frag.hits()[i];
00285             Target* t = m.mapped_targ;
00286             if (targ_set.count(t) == 0)
00287             {
00288                 t->lock();

00290             }
00291             likelihoods[i] = t->log_likelihood(m, first_round);
00292             masses[i] = t->mass();
00293             variances[i] = t->mass_var();
00294             total_likelihood = log_sum(total_likelihood, likelihoods[i]);
00295             total_mass = log_sum(total_mass, masses[i]);
00296             total_variance = log_sum(total_variance, variances[i]);
00297             num_solveable += t->solveable();
00298         }
00299     }
00300     else
00301     {
00302         Target* t = frag.hits()[0]->mapped_targ;
00303         t->lock();
00304         targ_set.insert(t);
00305         total_likelihood = likelihoods[0];
00306     }
00307     
00308     assert(!islzero(total_likelihood));
00309     
00310     // merge bundles
00311     Bundle* bundle = frag.hits()[0]->mapped_targ->bundle();
00312     if (first_round)
00313     {
00314         bundle->incr_counts();
00315     }
00316     
00317     // normalize marginal likelihoods
00318     for(size_t i = 0; i < frag.num_hits(); ++i)
00319     {
00320         FragHit& m = *frag.hits()[i];
00321         Target* t  = m.mapped_targ;
00322  
00323         bundle = targ_table->merge_bundles(bundle, t->bundle());
00324         
00325         double p = likelihoods[i]-total_likelihood;
00326         double mass_t = mass_n + p;
00327         
00328         double v = HUGE_VAL;
00329         if (frag.num_hits() > 1)
00330             v = log_sum(variances[i] - 2*total_mass, total_variance + 2*masses[i] - 4*total_mass);
00331         
00332         assert(!isnan(v));
00333         assert(!(isnan(mass_t)||isinf(mass_t)));
00334         
00335         m.probability = sexp(p);
00336         
00337         assert(!isinf(m.probability));
00338         
00339         t->add_mass(p, v, mass_n);
00340         
00341         // update parameters
00342         if (first_round)
00343         {
00344             t->incr_counts(frag.num_hits()==1);
00345             if (!t->solveable() && num_solveable == frag.num_hits()-1)
00346             {
00347                 t->solveable(true);
00348             }
00349             if (!burned_out)
00350             {
00351                 if (m.pair_status() == PAIRED)
00352                     (globs.fld)->add_val(m.length(), mass_t);
00353                 if (globs.bias_table)
00354                     (globs.bias_table)->update_observed(m, mass_t);
00355                 if (globs.mismatch_table)
00356                     (globs.mismatch_table)->update(m, mass_t);
00357             }
00358         }
00359         
00360         if (calc_covar && (last_round || online_additional))
00361         {
00362             for(size_t j = i+1; j < frag.num_hits(); ++j)
00363             {
00364                 const FragHit& m2 = *frag.hits()[j];
00365                 double p2 = likelihoods[j]-total_likelihood;
00366                 if (sexp(p2) == 0)
00367                     continue;
00368                 
00369                 double covar = p + p2;
00370                 if ((first_round && last_round) || online_additional)
00371                     covar += 2*mass_n;
00372                 
00373                 targ_table->update_covar(m.targ_id, m2.targ_id, covar); 
00374             }
00375         }
00376     }
00377     
00378     foreach (Target* t, targ_set)
00379     {
00380         t->unlock();
00381     }
