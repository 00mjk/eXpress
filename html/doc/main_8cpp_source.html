00177 void process_fragment(double mass_n, Fragment* frag_p, FLD* fld, BiasBoss* bias_table, MismatchTable* mismatch_table, TranscriptTable* trans_table)
00178 {
00179     Fragment& frag = *frag_p;
00180     
00181     assert(frag.num_maps());
00182     
00183     frag.maps()[0]->mapped_trans->incr_bundle_counts();
00184     
00185     if (frag.num_maps()==1)
00186     // maps to a single location
00187     {
00188         const FragMap& m = *frag.maps()[0];
00189         Transcript* t  = m.mapped_trans;
00190         if (!t)
00191         {
00192             cerr << "ERROR: Transcript " << m.name << " not found in reference fasta.";
00193             exit(1);
00194         }
00195         
00196         //update parameters
00197         t->add_mass(0, mass_n);
00198         if (m.pair_status() == PAIRED)
00199                 fld->add_val(m.length(), mass_n);
00200         
00201         if (bias_table)
00202             bias_table->update_observed(m, mass_n - t->mass() + log((double)t->unbiased_effective_length()));
00203         if (mismatch_table)
00204             mismatch_table->update(m, mass_n);
00205        
00206         delete frag_p;
00207         return;
00208     }
00209     
00210     // maps to multiple locations
00211     
00212     // calculate marginal likelihoods
00213     vector<double> likelihoods(frag.num_maps());
00214     double total_likelihood = 0.0;
00215     for(size_t i = 0; i < frag.num_maps(); ++i)
00216     {
00217         const FragMap& m = *frag.maps()[i];
00218         Transcript* t = m.mapped_trans;
00219         likelihoods[i] = t->log_likelihood(m);
00220         total_likelihood = (i) ? log_sum(total_likelihood, likelihoods[i]):likelihoods[i];
00221     }
00222     if (sexp(total_likelihood) == 0)
00223     {
00224         delete frag_p;
00225         return;
00226     }
00227 
00228     // normalize marginal likelihoods
00229     TransID bundle_rep = trans_table->get_trans_rep(frag.maps()[0]->trans_id);
00230     for(size_t i = 0; i < frag.num_maps(); ++i)
00231     {
00232         const FragMap& m = *frag.maps()[i];
00233         Transcript* t  = m.mapped_trans;
00234         double p = likelihoods[i]-total_likelihood;
00235         double mass_t = mass_n + p;
00236         
00237         if (i > 0)
00238         {
00239             TransID m_rep = trans_table->get_trans_rep(m.trans_id);
00240             if (m_rep != bundle_rep)
00241                 bundle_rep = trans_table->merge_bundles(bundle_rep, m_rep);
00242         }
00243         
00244         if (sexp(p) == 0)
00245             continue;
00246         
00247         // update parameters
00248         t->add_mass(p, mass_n);
00249         
00250         if (m.pair_status() == PAIRED)
00251             fld->add_val(m.length(), mass_t);
00252  
00253         if (bias_table)
00254             bias_table->update_observed(m, mass_t - t->mass() + log((double)t->unbiased_effective_length()));
00255         if (mismatch_table)
00256             mismatch_table->update(m, mass_t);
00257         
00258         if (calc_covar)
00259         {
00260             for(size_t j = i+1; j < frag.num_maps(); ++j)
00261             {
00262                 const FragMap& m2 = *frag.maps()[j];
00263                 double p2 = likelihoods[j]-total_likelihood;
00264                 if (sexp(p2) == 0)
00265                     continue;
00266                 
00267                 double covar = 2*mass_n + p + p2;
00268                 trans_table -> update_covar(m.trans_id, m2.trans_id, covar); 
00269             }
00270         }
00271     }
00272         
00273     delete frag_p;
00274 }
