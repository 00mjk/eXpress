00269 void process_fragment(Fragment* frag_p)
00270 {
00271     Fragment& frag = *frag_p;
00272     const Library& lib = *frag.lib();
00273     
00274     frag.sort_hits();
00275     double mass_n = frag.mass();
00276     
00277     assert(frag.num_hits());
00278         
00279     // calculate marginal likelihoods
00280     vector<double> likelihoods(frag.num_hits(),0);
00281     vector<double> masses(frag.num_hits(),0);
00282     vector<double> variances(frag.num_hits(), 0);
00283     double total_likelihood = HUGE_VAL;
00284     double total_mass = HUGE_VAL;
00285     double total_variance = HUGE_VAL;
00286     size_t num_solvable = 0;
00287     
00288     boost::unordered_set<Target*> targ_set;
00289 
00290     if (frag.num_hits()>1)
00291     {
00292         for(size_t i = 0; i < frag.num_hits(); ++i)
00293         {
00294             
00295             const FragHit& m = *frag.hits()[i];
00296             Target* t = m.mapped_targ;
00297             if (targ_set.count(t) == 0)
00298             {
00299                 t->lock();
00300                 targ_set.insert(t);
00301             }
00302             likelihoods[i] = t->log_likelihood(m, first_round);
00303             masses[i] = t->mass();
00304             variances[i] = t->mass_var();
00305             total_likelihood = log_sum(total_likelihood, likelihoods[i]);
00306             total_mass = log_sum(total_mass, masses[i]);
00307             total_variance = log_sum(total_variance, variances[i]);
00308             num_solvable += t->solvable();
00309         }
00310     }
00311     else
00312     {
00313         Target* t = frag.hits()[0]->mapped_targ;
00314         t->lock();
00315         targ_set.insert(t);
00316         total_likelihood = likelihoods[0];
00317     }
00318     
00319     assert(!islzero(total_likelihood));
00320     
00321     // merge bundles
00322     Bundle* bundle = frag.hits()[0]->mapped_targ->bundle();
00323     if (first_round)
00324     {
00325         bundle->incr_counts();
00326     }
00327     
00328     // normalize marginal likelihoods
00329     for(size_t i = 0; i < frag.num_hits(); ++i)
00330     {
00331         FragHit& m = *frag.hits()[i];
00332         Target* t  = m.mapped_targ;
00333  
00334         bundle = lib.targ_table->merge_bundles(bundle, t->bundle());
00335         
00336         double p = likelihoods[i]-total_likelihood;
00337         double v = HUGE_VAL;
00338         if (frag.num_hits() > 1)
00339             v = log_sum(variances[i] - 2*total_mass, total_variance + 2*masses[i] - 4*total_mass);
00340         
00341         assert(!isnan(v));
00342         assert(!(isnan(p)||isinf(p)));
00343         
00344         m.probability = sexp(p);
00345         
00346         assert(!isinf(m.probability));
00347         
00348         t->add_mass(p, v, mass_n);
00349         
00350         // update parameters
00351         if (first_round)
00352         {
00353             t->incr_counts(frag.num_hits()==1);
00354             if (!t->solvable() && num_solvable == frag.num_hits()-1)
00355             {
00356                 t->solvable(true);
00357             }
00358             if ((!burned_out || edit_detect) && lib.mismatch_table)
00359                     (lib.mismatch_table)->update(m, p, lib.mass_n);
00360             if (!burned_out)
00361             {
00362                 if (m.pair_status() == PAIRED)
00363                     (lib.fld)->add_val(m.length(), p+lib.mass_n);
00364                 if (lib.bias_table)
00365                     (lib.bias_table)->update_observed(m, p+lib.mass_n);
00366             }
00367         }
00368         
00369         if (calc_covar && (last_round || online_additional))
00370         {
00371             for(size_t j = i+1; j < frag.num_hits(); ++j)
00372             {
00373                 const FragHit& m2 = *frag.hits()[j];
00374                 double p2 = likelihoods[j]-total_likelihood;
00375                 if (sexp(p2) == 0)
00376                     continue;
00377                 
00378                 double covar = p + p2;
00379                 if ((first_round && last_round) || online_additional)
00380                     covar += 2*mass_n;
00381                 
00382                 lib.targ_table->update_covar(m.targ_id, m2.targ_id, covar); 
00383             }
00384         }
00385     }
00386     
00387     foreach (Target* t, targ_set)
00388     {
00389         t->unlock();
00390     }
00391 }
