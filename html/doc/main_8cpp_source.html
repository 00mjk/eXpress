00190 void process_fragment(double mass_n, Fragment* frag_p, TranscriptTable* trans_table, const Globals& globs)
00191 {
00192     Fragment& frag = *frag_p;
00193     
00194     assert(frag.num_hits());
00195     
00196     Bundle* bundle = frag.hits()[0]->mapped_trans->bundle();
00197     if (iteration != FIRST)
00198         bundle->incr_counts();
00199     
00200     if (frag.num_hits()==1)
00201     // hits to a single location
00202     {
00203         FragHit& m = *frag.hits()[0];
00204         Transcript* t  = m.mapped_trans;
00205         
00206         m.probability = 1.0;
00207         
00208         //update parameters
00209         if (iteration == FIRST || iteration == ONLY)
00210         {
00211             t->add_mass(0, mass_n);
00212             t->incr_uniq_counts();
00213             if (m.pair_status() == PAIRED)
00214                     (globs.fld)->add_val(m.length(), mass_n);
00215             
00216             if (globs.bias_table)
00217                 (globs.bias_table)->update_observed(m, mass_n - t->mass() + log((double)t->unbiased_effective_length()));
00218             if (globs.mismatch_table)
00219                 (globs.mismatch_table)->update(m, mass_n);
00220         }
00221         else
00222         {
00223             t->add_prob_count(1.0);
00224         }
00225         return;
00226     }
00227     
00228     // hits to multiple locations
00229     
00230     // calculate marginal likelihoods
00231     vector<double> likelihoods(frag.num_hits());
00232     double total_likelihood = 0.0;
00233     for(size_t i = 0; i < frag.num_hits(); ++i)
00234     {
00235         const FragHit& m = *frag.hits()[i];
00236         Transcript* t = m.mapped_trans;
00237         likelihoods[i] = t->log_likelihood(m);
00238         total_likelihood = (i) ? log_sum(total_likelihood, likelihoods[i]):likelihoods[i];
00239     }
00240     if (sexp(total_likelihood) == 0)
00241     {
00242         return;
00243     }
00244     
00245     // merge bundles
00246     for(size_t i = 1; i < frag.num_hits(); ++i)
00247     {
00248         bundle = trans_table->merge_bundles(bundle, frag.hits()[i]->mapped_trans->bundle());
00249     }
00250 
00251     // normalize marginal likelihoods
00252     for(size_t i = 0; i < frag.num_hits(); ++i)
00253     {
00254         FragHit& m = *frag.hits()[i];
00255         Transcript* t  = m.mapped_trans;
00256         double p = likelihoods[i]-total_likelihood;
00257         double mass_t = mass_n + p;
00258                
00259         m.probability = sexp(p);
00260         if (m.probability == 0)
00261             continue;
00262         assert(!isinf(m.probability));
00263         
00264         // update parameters
00265         if (iteration == FIRST || iteration == ONLY)
00266         {
00267 
00268             t->add_mass(p, mass_n);
00269             
00270             if (m.pair_status() == PAIRED)
00271                 (globs.fld)->add_val(m.length(), mass_t);
00272      
00273             if (globs.bias_table)
00274                 (globs.bias_table)->update_observed(m, mass_t - t->mass() + log((double)t->unbiased_effective_length()));
00275             if (globs.mismatch_table)
00276                 (globs.mismatch_table)->update(m, mass_t);
00277         }
00278         else
00279         {
00280             t->add_prob_count(m.probability);
00281         }
00282         
00283         if (calc_covar && (iteration == ONLY || iteration == LAST))
00284         {
00285             for(size_t j = i+1; j < frag.num_hits(); ++j)
00286             {
00287                 const FragHit& m2 = *frag.hits()[j];
00288                 double p2 = likelihoods[j]-total_likelihood;
00289                 if (sexp(p2) == 0)
00290                     continue;
00291                 
00292                 double covar = p + p2;
00293                 if (iteration == ONLY)
00294                     covar += 2*mass_n;
00295                 
00296                 trans_table->update_covar(m.trans_id, m2.trans_id, covar); 
00297             }
00298         }
00299     }
